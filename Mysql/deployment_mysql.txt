  name: [nombre del pod]
  namespace: [nombre del namespace]
  labels:
    app: [nombre del pod]
spec:
  replicas: 1
  selector:
    matchLabels:
      app: [nombre del pod]
  template:
    metadata:
      labels:
        app: [nombre del pod]
    spec:
      containers:
        - name: [nombre del pod]
          image: mysql:5.7
          ports:
            - containerPort: 3306
          volumeMounts:
            - mountPath: "/var/lib/mysql"
              subPath: "mysql"
              name: [nombre del volumen]
          env:
            - name: MYSQL_ROOT_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: [nombre del secreto]
                  key: ROOT_PASSWORD
      volumes:
        - name: [nombre del volumen]
          persistentVolumeClaim:
            claimName: [nombre del volumen]
