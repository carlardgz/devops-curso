apiVersion: v1
kind: Namespace
metadata:
  name: aplicacioncurso

---
apiVersion: v1
kind: Secret
metadata:
  name: mysecret
  namespace: aplicacioncurso
type: Opaque
data:
  ROOT_PASSWORD: bWktY29udHJhc2XDsWEtZGVsLWN1cnNv

---

apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: volumenmysql
  namespace: aplicacioncurso
spec:
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 1Gi
---

apiVersion: apps/v1
kind: Deployment
metadata:
  name: mysqlcurso
  namespace: aplicacioncurso
  labels:
    app: mysqlcurso
spec:
  replicas: 1
  selector:
    matchLabels:
      app: mysqlcurso
  template:
    metadata:
      labels:
        app: mysqlcurso
    spec:
      containers:
        - name: mysqlcurso
          image: mysql:5.7
          ports:
            - containerPort: 3306
          volumeMounts:
            - mountPath: "/var/lib/mysql"
              subPath: "mysql"
              name: volumenmysql
          env:
            - name: MYSQL_ROOT_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: mysecret
                  key: ROOT_PASSWORD
      volumes:
        - name: volumenmysql
          persistentVolumeClaim:
            claimName: volumenmysql

---
apiVersion: v1
kind: Service
metadata:
  name: servicecurso
  namespace: aplicacioncurso
spec:
  selector:
    app: servicecurso
  ports:
  - protocol: TCP
    port: 3306
    targetPort: 3306
